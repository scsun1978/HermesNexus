def _cmd_uptime(self):
    uptime_seconds = int(time.time() - self.uptime_start)
    hours = uptime_seconds // 3600
    minutes = (uptime_seconds % 3600) // 60
    return f"up {hours}:{minutes:02d}, {self.load_average[0]:.2f}"
